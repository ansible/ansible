"""

- action: shell echo hi

# most commonly: 

- copy: src=a dest=b 

 

# legacy form 

- action: copy src=a dest=b 

 

# complex args form, for passing structured data 

- copy: 

src: a 

dest: b 

 

# gross, but technically legal 

- action: 

module: copy 

args: 

src: a 

dest: b

# will act as 'defaults' and will be overridden by any args specified
